```
#include <unistd.h>    // fork(), getpid(), getppid(), exec()
#include <sys/types.h> // pid_t
#include <sys/wait.h>  // wait(), waitpid()
#include <stdio.h>     // printf(), perror()
#include <stdlib.h>    // exit()
#include <signal.h> // para kill() y SIGKILL


void calcularBiciestos();
void calcularPrimos();

int main(){

    int p1 = fork();

    if(p1==0){
        // proceso  1
        calcularPrimos();
    }else{
        int p2 = fork();

        if(p2 ==0){
            // proceso  2
            calcularBiciestos();
        }else{
            // proceso padre
            sleep(4);
            kill(p1, SIGKILL);
            sleep(8);
            kill(p2, SIGKILL);
            printf("Procesos hijos finalizados");
            exit(0);
        }

    }

    return 0;
}

void calcularBiciestos(){
    int i = 0;
    while(1){
        
        if((i%4 == 0)&& ((i%100!=0)|| (( i%100==0 ) &&(i%400 == 0 )))){
           printf("Hijo 2: %d\n", i);

        }
        i++;
    }
}

void calcularPrimos(){
    int i = 2 ; 
    while(1) {
            int esPrimo = 1;
            for (int j = 2; j < i; j++) {
                if (i % j == 0) {
                    esPrimo = 0;
                    break;
                }
            }
            if (esPrimo) {
                printf("Hijo 1: %d\n", i);
            }
            i++;
    }
}
```
